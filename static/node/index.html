The spread operator


    const cities = ["Kampala", "Nairobi", "Lagos"];
    console.log(...cities);
    
    //output
    Kampala Nairobi Lagos

Template literals


    const student = {
      name: 'John Kagga',
      city: 'Kampala'
    };
    
    let message = 'Hello ' + student.name + ' from ' + student.city;
    console.log(message);
    //output: Hello John Kagga from Kampala
    
    // ---------------- vs ------------------
    
    let message = `Hello ${student.name} from ${student.city}`;
    console.log(message);
    //output: Hello John Kagga from Kampala

Default function parameters


    function greet(name = 'Fellow', greeting = 'Welcome') {
      return `${greeting} ${name}`;
    }
    
    console.log(greet()); // Welcome Fellow
    console.log(greet('Kagga')); // Welcome Kagga
    console.log(greet('Mike', 'Hi')); // Hi Mike

Destructuring


    const points = [20, 30, 40];
    const x = points[0];
    const y = points[1];
    const z = points[2];
    
    console.log(x, y, z); //output: 20 30 40
    
    // ---------------- vs ------------------
    
    const points = [20, 30, 40];
    const [x, y, z] = points;
    
    console.log(x, y, z); //output: 20 30 40

Object literal Shorthand


    let type = 'Toyota';
    let color = 'Silver';
    let model = 2007;
    
    const car = {
      type: type,
      color: color,
      model: model
    };
    
    // ---------------- vs ------------------
    
    const car = {type, color, model};
